file(GLOB OPTION_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*)

list(REMOVE_ITEM OPTION_FILES ${CMAKE_CURRENT_SOURCE_DIR}/CMakeLists.txt)

foreach(FILE_PATH IN LISTS OPTION_FILES)
    get_filename_component(FILENAME ${FILE_PATH} NAME)
    configure_file(${FILE_PATH} ${CMAKE_ARCHIVE_OUTPUT_DIRECTORY}/${FILENAME} COPYONLY)
endforeach()
